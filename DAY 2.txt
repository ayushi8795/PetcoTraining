3)Count all lower case, upper case, digits, and special symbols from a given string
def check(str1):
    ccount=digit=symbol = 0

    for i in str1:
        if i >= "a" and i <= "z":
            ccount = ccount + 1
        elif i >= "0" and i<= "9":
            digit = digit + 1 
        else:
            symbol = symbol + 1 
            
    print('chars = {}'.format(ccount))
    print('digit = {}'.format(digit))
    print('symbol = ',symbol)
